        \psline[linewidth=0.05]{cc-cc}(.1,-.1)(-.1,.1)
      }
    }
    % area
    \psframe[linewidth=0.0001,fillstyle=hlines,hatchwidth=0.01,hatchcolor=black](0, 9)(#1,#2)
  }
